TYPE
	 correct_value_analog : REAL(4.0..20.0)(*zakres możliwej wartości sygnału z czujnika *);
END_TYPE

FUNCTION NORM : REAL;

VAR
	range_min : REAL;
	range_max : REAL;
	value : REAL CZY correct_value_analog;
END_VAR


NORM := (value-range_min)/(range_max-range_min);
 
END_FUNCTION

FUNCTION SCALE: REAL;
 VAR	
	rang_max : REAL;
	range_min : REAL;
	value
END_VAR
	SCALE :=  (value *(range_max-range_min)) + range_min;
